1. Create a simple AWT application that displays a button with FlowLayout manager.

Code:

import java.awt.*;
import java.awt.event.*;

public class ButtonFlowLayout {
    public static void main(String[] args) {
        Frame frame = new Frame("Button with FlowLayout");
        Button button = new Button("Click Me");
        frame.setLayout(new FlowLayout());
        frame.add(button);
        frame.setSize(300, 200);
        frame.setVisible(true);
    }
}

3. Create a simple AWT application that displays multiple buttons with BorderLayout manager (North, East, West, South, Center).

Code:

import java.awt.*;
import java.awt.event.*;

public class ButtonBorderLayout {
    public static void main(String[] args) {
        Frame frame = new Frame("Multiple Buttons with BorderLayout");
        Button button1 = new Button("North");
        Button button2 = new Button("East");
        Button button3 = new Button("West");
        Button button4 = new Button("South");
        Button button5 = new Button("Center");

        frame.setLayout(new BorderLayout());
        frame.add(button1, BorderLayout.NORTH);
        frame.add(button2, BorderLayout.EAST);
        frame.add(button3, BorderLayout.WEST);
        frame.add(button4, BorderLayout.SOUTH);
        frame.add(button5, BorderLayout.CENTER);
        frame.setSize(400, 300);
        frame.setVisible(true);
    }
}

18. Create a simple AWT application to design a calculator that displays buttons, TextField with GridLayout manager.

Code:

import java.awt.*;
import java.awt.event.*;

public class Calculator {
    public static void main(String[] args) {
        Frame frame = new Frame("Calculator");
        TextField textField = new TextField();
        Button[] buttons = new Button[16];
        String[] buttonLabels = {"7", "8", "9", "/", "4", "5", "6", "*", "1", "2", "3", "-", "0", "=", "+", "C"};
        frame.setLayout(new GridLayout(4, 4));
        frame.add(textField);

        for (int i = 0; i < 16; i++) {
            buttons[i] = new Button(buttonLabels[i]);
            frame.add(buttons[i]);
        }

        frame.setSize(300, 400);
        frame.setVisible(true);
    }
}
