3. Java Program to Create a Button and Display Image in the Frame when Clicked

Code:

import java.awt.*;
import java.awt.event.*;
import javax.swing.*;

public class ButtonImageExample {
    public static void main(String[] args) {
        JFrame frame = new JFrame("Button Image Example");
        JButton button = new JButton("Show Image");
        JLabel imageLabel = new JLabel();
        frame.setLayout(new FlowLayout());
        button.addActionListener(e -> {
            ImageIcon icon = new ImageIcon("path/to/image.jpg");
            imageLabel.setIcon(icon);
        });
        frame.add(button);
        frame.add(imageLabel);
        frame.setSize(400, 300);
        frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        frame.setVisible(true);
    }
}

Output:

When the "Show Image" button is clicked, an image appears in the frame (make sure to replace "path/to/image.jpg" with an actual image path).

9. Java Program to Create and Switch between Frames using Buttons

Code:

import java.awt.*;
import java.awt.event.*;
import javax.swing.*;

public class FrameSwitchExample {
    public static void main(String[] args) {
        JFrame frame1 = new JFrame("Frame 1");
        JFrame frame2 = new JFrame("Frame 2");

        JButton switchToFrame2 = new JButton("Switch to Frame 2");
        JButton switchToFrame1 = new JButton("Switch to Frame 1");

        switchToFrame2.addActionListener(e -> {
            frame1.setVisible(false);
            frame2.setVisible(true);
        });
        switchToFrame1.addActionListener(e -> {
            frame2.setVisible(false);
            frame1.setVisible(true);
        });

        frame1.setLayout(new FlowLayout());
        frame1.add(switchToFrame2);
        frame1.setSize(300, 200);
        frame1.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        frame1.setVisible(true);

        frame2.setLayout(new FlowLayout());
        frame2.add(switchToFrame1);
        frame2.setSize(300, 200);
        frame2.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
    }
}

Output:

Clicking the button switches between Frame 1 and Frame 2.

18. Java Program to Display Several Dots on the Screen Continuously

Code:

import java.awt.*;
import javax.swing.*;

public class DotsExample extends JPanel {
    public void paintComponent(Graphics g) {
        super.paintComponent(g);
        for (int i = 0; i < getWidth(); i += 20) {
            for (int j = 0; j < getHeight(); j += 20) {
                g.fillOval(i, j, 5, 5);
            }
        }
    }

    public static void main(String[] args) {
        JFrame frame = new JFrame("Dots Example");
        frame.add(new DotsExample());
        frame.setSize(400, 300);
        frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        frame.setVisible(true);
    }
}

Output:

A frame displays several dots arranged on the screen continuously.

19. Java Program to Validate the TextField for Only Entering Numbers using KeyListener

Code:

import javax.swing.*;
import java.awt.event.*;

public class TextFieldNumberValidation extends JFrame {
    JTextField textField;

    public TextFieldNumberValidation() {
        textField = new JTextField(20);
        textField.addKeyListener(new KeyAdapter() {
            public void keyTyped(KeyEvent e) {
                char c = e.getKeyChar();
                if (!Character.isDigit(c)) {
                    e.consume();
                }
            }
        });
        
        add(textField);
        setSize(300, 200);
        setLayout(new FlowLayout());
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        setVisible(true);
    }

    public static void main(String[] args) {
        new TextFieldNumberValidation();
    }
}

Output:

The text field allows only numeric input.

20. Java Program to Validate the TextField for Only Entering Email ID using KeyAdapter

Code:

import javax.swing.*;
import java.awt.event.*;

public class EmailValidationExample extends JFrame {
    JTextField textField;

    public EmailValidationExample() {
        textField = new JTextField(20);
        textField.addKeyListener(new KeyAdapter() {
            public void keyReleased(KeyEvent e) {
                String text = textField.getText();
                if (text.matches("^[\\w.-]+@[\\w.-]+\\.[a-zA-Z]{2,6}$")) {
                    textField.setForeground(Color.BLACK);
                } else {
                    textField.setForeground(Color.RED);
                }
            }
        });
        
        add(textField);
        setSize(300, 200);
        setLayout(new FlowLayout());
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        setVisible(true);
    }

    public static void main(String[] args) {
        new EmailValidationExample();
    }
}

Output:

The text field displays the entered email ID in red if invalid and black if valid.
