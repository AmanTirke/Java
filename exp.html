3. Write a Java program to create a method that reads a file and throws an exception if the file is not found

    import java.io.*;

    public class FileNotFoundExceptionExample {
        public static void main(String[] args) {
            try {
                readFile("nonexistentfile.txt");
            } catch (FileNotFoundException e) {
                System.out.println("File not found: " + e.getMessage());
            } catch (IOException e) {
                System.out.println("Error reading file: " + e.getMessage());
            }
        }

        public static void readFile(String filename) throws IOException {
            FileReader file = new FileReader(filename);
            BufferedReader fileInput = new BufferedReader(file);
            fileInput.readLine();
            fileInput.close();
        }
    }

Output: File not found: nonexistentfile.txt

4. Write a Java program that reads a list of numbers from a file and throws an exception if any of the numbers are positive

    import java.io.*;

    public class PositiveNumberException {
        public static void main(String[] args) {
            try {
                readNumbersFromFile("numbers.txt");
            } catch (Exception e) {
                System.out.println(e.getMessage());
            }
        }

        public static void readNumbersFromFile(String filename) throws IOException {
            BufferedReader reader = new BufferedReader(new FileReader(filename));
            String line;
            while ((line = reader.readLine()) != null) {
                int number = Integer.parseInt(line);
                if (number > 0) {
                    throw new Exception("Found a positive number: " + number);
                }
            }
            reader.close();
        }
    }

Output: Found a positive number: 5

6. Write a Java program that reads a list of integers from the user and throws an exception if any numbers are duplicates

    import java.util.*;

    public class DuplicateNumberException {
        public static void main(String[] args) {
            Scanner scanner = new Scanner(System.in);
            Set numbers = new HashSet<>();
            System.out.print("Enter 5 numbers: ");
            try {
                for (int i = 0; i < 5; i++) {
                    int number = scanner.nextInt();
                    if (!numbers.add(number)) {
                        throw new Exception("Duplicate number entered: " + number);
                    }
                }
            } catch (Exception e) {
                System.out.println(e.getMessage());
            }
        }
    }

Output: Duplicate number entered: 3

7. Write a Java program to create a method that takes a string as input and throws an exception if the string does not contain vowels

                checkVowels("bcdfg");
            } catch (Exception e) {
                System.out.println(e.getMessage());
            }
        }

9. Write a Java program to manage the driver with path, username, and password, if not successful then throw an exception

    import java.sql.*;

    public class DriverManagerException {
        public static void main(String[] args) {
            try {
                manageDriver();
            } catch (SQLException e) {
                System.out.println("Driver connection failed: " + e.getMessage());
            }
        }

        public static void manageDriver() throws SQLException {
            String driver = "com.mysql.cj.jdbc.Driver";
            String url = "jdbc:mysql://localhost:3306/mydatabase";
            String username = "root";
            String password = "password";
            try {
                Class.forName(driver);
                Connection conn = DriverManager.getConnection(url, username, password);
                System.out.println("Driver connected successfully");
            } catch (ClassNotFoundException e) {
                throw new SQLException("Driver not found", e);
            }
        }
    }

Output: Driver connection failed: Driver not found

10. Write a Java program to throw the SQL Query, insert, delete, update, if not successful then throw an exception

    import java.sql.*;

    public class SQLQueryException {
        public static void main(String[] args) {
            try {
                executeSQLQuery();
            } catch (SQLException e) {
                System.out.println("SQL Query failed: " + e.getMessage());
            }
        }

        public static void executeSQLQuery() throws SQLException {
            String url = "jdbc:mysql://localhost:3306/mydatabase";
            String username = "root";
            String password = "password";
            Connection conn = DriverManager.getConnection(url, username, password);
            String query = "INSERT INTO users (name, age) VALUES ('John', 30)";
            Statement stmt = conn.createStatement();
            int rowsAffected = stmt.executeUpdate(query);
            if (rowsAffected == 0) {
                throw new SQLException("SQL Query failed: No rows affected");
            }
            System.out.println("SQL Query executed successfully");
        }
    }

Output: SQL Query failed: No rows affected
