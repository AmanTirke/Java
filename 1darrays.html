1. Program to copy all elements of one array into another array

    public class ArrayCopyExample {
        public static void main(String[] args) {
            int[] arr1 = {1, 2, 3, 4, 5};
            int[] arr2 = new int[arr1.length];
            System.arraycopy(arr1, 0, arr2, 0, arr1.length);
            System.out.println("Array 2 elements: ");
            for (int num : arr2) {
                System.out.print(num + " ");
            }
        }
    }

Output: Array 2 elements: 1 2 3 4 5

2. Program to find the frequency of each element in the array

    import java.util.*;

    public class FrequencyOfElements {
        public static void main(String[] args) {
            int[] arr = {1, 2, 2, 3, 3, 3, 4};
            Map freqMap = new HashMap<>();
            for (int num : arr) {
                freqMap.put(num, freqMap.getOrDefault(num, 0) + 1);
            }
            for (Map.Entry entry : freqMap.entrySet()) {
                System.out.println(entry.getKey() + ": " + entry.getValue());
            }
        }
    }

Output: 1: 1
2: 2
3: 3
4: 1

3. Program to left rotate the elements of an array

    public class LeftRotateArray {
        public static void main(String[] args) {
            int[] arr = {1, 2, 3, 4, 5};
            int n = 2; // number of rotations
            int[] rotatedArr = new int[arr.length];
            for (int i = 0; i < arr.length; i++) {
                rotatedArr[i] = arr[(i + n) % arr.length];
            }
            System.out.println("Left rotated array: " + Arrays.toString(rotatedArr));
        }
    }

Output: Left rotated array: [3, 4, 5, 1, 2]

    }

6. Program to print the elements of an array in reverse order

    public class ReverseArray {
        public static void main(String[] args) {
            int[] arr = {1, 2, 3, 4, 5};
            System.out.println("Array elements in reverse order:");
            for (int i = arr.length - 1; i >= 0; i--) {
                System.out.print(arr[i] + " ");
            }
        }
    }

Output: Array elements in reverse order: 5 4 3 2 1

22. Program to Print Odd and Even Numbers from an array

    public class OddEvenNumbers {
        public static void main(String[] args) {
            int[] arr = {1, 2, 3, 4, 5, 6};
            System.out.print("Odd numbers: ");
            for (int num : arr) {
                if (num % 2 != 0) {
                    System.out.print(num + " ");
                }
            }
            System.out.println();
            System.out.print("Even numbers: ");
            for (int num : arr) {
                if (num % 2 == 0) {
                    System.out.print(num + " ");
                }
            }
        }
    }

Output: Odd numbers: 1 3 5
Even numbers: 2 4 6
